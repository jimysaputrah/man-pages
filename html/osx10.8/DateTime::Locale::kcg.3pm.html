NAME
       DateTime::Locale::kcg

SYNOPSIS
	 use DateTime;

	 my $dt = DateTime->now( locale => 'kcg' );
	 print $dt->month_name();

DESCRIPTION
       This is the DateTime locale package for Tyap.

DATA
