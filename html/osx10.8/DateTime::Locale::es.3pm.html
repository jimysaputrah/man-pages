NAME
       DateTime::Locale::es

SYNOPSIS
	 use DateTime;

	 my $dt = DateTime->now( locale => 'es' );
	 print $dt->month_name();

DESCRIPTION
       This is the DateTime locale package for Spanish.

DATA
