NAME
       DateTime::Locale::id

SYNOPSIS
	 use DateTime;

	 my $dt = DateTime->now( locale => 'id' );
	 print $dt->month_name();

DESCRIPTION
       This is the DateTime locale package for Indonesian.

DATA
